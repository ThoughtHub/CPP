void select_sort(string array[],int n)              
{
    int i,j,k,
    string t;
    for(i=0; i<n-1; i++)
    {
        k=i;  
        for(j=i+1; j<n; j++)
            if(array[j]<array[k])
                k=j;   
        t=array[k]; 
        array[k]=array[i]; 
        array[i]=t;
    }
    return;
}
